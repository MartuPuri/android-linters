import android.os.Bundle;
import android.app.Fragment;
import android.util.Log;
import java.util.HashMap;
import android.annotation.SuppressLint;

public class TestIgnoreRestoreAStateWithDifferentType extends Fragment {

	private static final String KEY_INT = "KEY_INT";
	private static final String KEY_CHAR = "KEY_CHAR";
	private double doubleValue;
	private int intValue;
	private char charValue;

	@Override
	public void onCreate(final Bundle savedInstanceState) {
		super.onCreate(savedInstanceState);
		if (savedInstanceState != null) {
			intValue = savedInstanceState.getChar(KEY_CHAR);
		}
	}

	@Override
	@SuppressLint("invalidType")
	public void onActivityCreated(final Bundle savedInstanceState) {
		super.onActivityCreated(savedInstanceState);
		if (savedInstanceState != null) {
			doubleValue = savedInstanceState.getInt(KEY_INT);
		}
	}

	@Override
	public void onSaveInstanceState(final Bundle outState) {
		outState.putChar(KEY_CHAR, charValue);
		outState.putInt(KEY_INT, intValue);
		super.onSaveInstanceState(outState);
	}
}
